q = 40      # Capacit maximale du camion
omega = 50  # Cot fixe par camion utilis

def fitness(state, solution, distance_matrix):
    """
    Calcule la fonction objectif d'une solution donne.
    
    Args:
        state: Dictionnaire contenant les informations du problme
        solution: Liste d'entiers reprsentant l'ordre de visite des clients
        distance_matrix: Matrice des distances entre les points
        
    Returns:
        Valeur de la fonction objectif (cot total)
    """
    orders = state["orders"]

    delivered = set()
    capacity = q
    total_distance = 0
    
    penalty_count = 0  # Compteur de violations de contraintes
    penalty_factor = 1000  # Multiplicateur de pnalit

    trucks = 0

    for i in range(1, len(solution)):
        prev, curr = solution[i-1], solution[i]
        
        # Vrifier si le chemin existe
        if distance_matrix[prev, curr] == float("inf"):
            penalty_count += 6  # Le chemin n'existe pas

        total_distance += distance_matrix[prev, curr]
        
        # Si on est chez un client, livrer sa commande
        if curr != 0:  
            if capacity < orders[curr]:
                penalty_count += (orders[curr] - capacity + 1) // 2  # Capacit insuffisante
            
            capacity -= orders[curr]
            delivered.add(curr)
        
        # Si on est de retour au dpt, recharger le camion
        if curr == 0:
            trucks += 1
            capacity = q

    # S'assurer que toutes les commandes sont livres
    if len(delivered) < len(orders) - 1:  # En excluant le dpt
        penalty_count += 5 * (len(orders) - 1 - len(delivered))  # Certaines commandes n'ont pas t livres

    # S'assurer que le chemin commence et se termine au dpt
    if solution[0] != 0 or solution[-1] != 0:
        penalty_count += 2  # La route ne commence/finit pas au dpt

    # Score final: distance totale + cot des camions + pnalits
    trucks_value = trucks * omega
    penalty_value = penalty_count * penalty_factor
    return total_distance + trucks_value + penalty_value

def compute_distance_matrix(state):
    """
    Calcule la matrice des distances entre tous les points.
    
    Args:
        state: Dictionnaire contenant les positions des points
        
    Returns:
        Matrice des distances
    """
    coords = state["position"]
    n = len(coords)

    dist_matrix = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            dist_matrix[i, j] = np.sqrt((coords[i][0] - coords[j][0])**2 + (coords[i][1] - coords[j][1])**2)
    return dist_matrix

def count_trucks(solution):
    """
    Compte le nombre de camions utiliss (nombre de retours au dpt).
    
    Args:
        solution: Liste reprsentant une solution
        
    Returns:
        Nombre de camions utiliss
    """
    return solution.count(0) - 1  # On retire le premier 0 (dpart)

def is_valid_solution(solution, orders, capacity):
    """
    Vrifie que chaque camion respecte sa capacit.
    
    Args:
        solution: Liste reprsentant une solution
        orders: Liste des commandes des clients
        capacity: Capacit maximale du camion
        
    Returns:
        Boolen indiquant si la solution est valide
    """
    total_load = 0
    for node in solution:
        if node == 0:  # Retour au dpt -> Rinitialisation de la charge
            total_load = 0
        else:
            total_load += orders[node]
            if total_load > capacity:
                return False  # Dpassement de la capacit
    return True

L'algorithme de recherche tabou est une mtaheuristique qui permet d'viter les optima locaux en interdisant temporairement (mettant "tabou") certains mouvements qui pourraient ramener des solutions dj visites.

Notre implmentation :

  1. Commence avec une solution initiale.
  2. Explore le voisinage en effectuant des changes entre les positions des clients.
  3. Choisit le meilleur voisin non tabou (ou utilise un critre d'aspiration).
  4. Met jour la liste tabou et continue jusqu' atteindre le nombre maximum d'itrations.

La taille de la liste tabu_list varie au cours des itrations, mais elle est principalement contrle par le paramtre tabu_tenure. Voici son volution :

  1. Ajout d'un mouvement tabou :

    • Lorsqu'un mouvement (i, j) est slectionn pour une nouvelle solution, il est ajout tabu_list :
      tabu_list.append(move)

  2. Limitation de la taille de la liste tabou :

    • Si la taille de tabu_list dpasse tabu_tenure, le plus ancien mouvement est supprim (stratgie FIFO) :
      if len(tabu_list) > tabu_tenure:
    tabu_list.pop(0)

  3. Rinitialisation conditionnelle :

    • Si une meilleure solution globale est trouve, la liste tabou est vide :
      tabu_list = []

Conclusion :

  • En gnral, tabu_list ne dpasse pas tabu_tenure lments, sauf lorsqu'elle est vide aprs l'amlioration d'une solution.
  • Si aucune meilleure solution globale n'est trouve pendant max_iter, la taille de tabu_list sera au plus tabu_tenure aprs quelques itrations.
  • Si une nouvelle meilleure solution est trouve, tabu_list est rinitialise et repart de zro.

def tabu_search(state, initial_solution, max_iter, tabu_tenure):
    """
    Implmentation de la recherche tabou pour le problme de tournes de vhicules.
    
    Args:
        state: Dictionnaire contenant les informations du problme
        initial_solution: Solution initiale
        max_iter: Nombre maximum d'itrations
        tabu_tenure: Taille de la liste tabou
        
    Returns:
        Tuple contenant la meilleure solution trouve et son cot
    """
    start_time = time.time()  # Dbut du chronomtre
    
    distance_matrix = compute_distance_matrix(state)
    best_solution = initial_solution.copy()
    current_solution = initial_solution.copy()
    best_cost = fitness(state, best_solution, distance_matrix)
    current_cost = best_cost
    tabu_list = []
    
    # Pour suivre l'volution de la recherche
    iteration_costs = []
    best_costs = []
    
    for iteration in range(max_iter):
        neighborhood = []

        # Gnrer des voisins avec des swaps
        for i in range(1, len(current_solution) - 1):  # Ignorer le dpt (0)
            for j in range(i+1, len(current_solution) - 1):
                if i != j:
                    new_solution = current_solution.copy()
                    new_solution[i], new_solution[j] = new_solution[j], new_solution[i]
                    
                    if is_valid_solution(new_solution, state['orders'], q):
                        cost = fitness(state, new_solution, distance_matrix)
                        move = (i, j)
                        neighborhood.append((new_solution, cost, move))
        
        if not neighborhood:
            print(f"Itration {iteration}: Aucun voisin valide trouv, arrt de la recherche.")
            break  # Si aucun voisin n'est valide, on arrte

        # Trier les voisins par cot
        neighborhood.sort(key=lambda x: x[1])

        # Slectionner le meilleur voisin non tabou (ou le meilleur voisin si aspiration)
        found_better = False
        for candidate, cost, move in neighborhood:
            if move not in tabu_list or cost < best_cost:  # Critre d'aspiration
                current_solution = candidate.copy()
                current_cost = cost
                
                if move not in tabu_list:
                    tabu_list.append(move)
                    if len(tabu_list) > tabu_tenure:
                        tabu_list.pop(0)
                
                if cost < best_cost:
                    best_solution = candidate.copy()
                    best_cost = cost
                
                found_better = True
                break
        
        if not found_better:
            # Si tous les mouvements sont tabou, on choisit le meilleur quand mme
            candidate, cost, move = neighborhood[0]
            current_solution = candidate.copy()
            current_cost = cost
            tabu_list.append(move)
            if len(tabu_list) > tabu_tenure:
                tabu_list.pop(0)
        
        # Enregistrer les cots pour l'analyse
        iteration_costs.append(current_cost)
        best_costs.append(best_cost)
        
       
    
    end_time = time.time()  # Fin du chronomtre
    execution_time = end_time - start_time
    
    
    # Retourner la meilleure solution, son cot, et les donnes d'analyse
    return best_solution, best_cost, iteration_costs, best_costs, execution_time

def plot_solution(state, solution, title=""):
    """
    Visualise la solution sur un graphique.
    
    Args:
        state: Dictionnaire contenant les positions des points
        solution: Liste reprsentant une solution
        title: Titre du graphique
    """
    positions = state["position"]
    depot = positions[0]
    clients = positions[1:]
    
    plt.figure(figsize=(10, 6))
    plt.scatter(*depot, c='red', marker='s', s=100, label='Dpt')
    for i, (x, y) in enumerate(clients, start=1):
        plt.scatter(x, y, c='blue', s=50)
        plt.text(x, y, f"{i}\n(q={state['orders'][i]})", fontsize=9, verticalalignment='bottom', horizontalalignment='right')
    
    colors = ['g', 'b', 'm', 'c', 'y', 'orange', 'purple']
    truck_count = 0
    current_color = colors[truck_count % len(colors)]
    
    for i in range(len(solution) - 1):
        start, end = solution[i], solution[i + 1]
        if start == 0 and end != 0:  # Nouveau camion qui quitte le dpt
            truck_count += 1
            current_color = colors[(truck_count-1) % len(colors)]
            plt.plot([positions[start][0], positions[end][0]], 
                    [positions[start][1], positions[end][1]], 
                    color=current_color, linewidth=2, marker='o',
                    label=f'Camion {truck_count}')
        else:
            plt.plot([positions[start][0], positions[end][0]], 
                    [positions[start][1], positions[end][1]], 
                    color=current_color, linewidth=2, marker='o')
    
    plt.legend()
    plt.grid(True, linestyle='--', alpha=0.7)
    plt.title(f"{title} - Nombre de camions : {truck_count}")
    plt.xlabel("Coordonne X")
    plt.ylabel("Coordonne Y")
    plt.show()

def plot_cost_evolution_interation(iteration_costs, best_costs):
    """
    Trace l'volution des cots au cours des itrations.
    
    Args:
        iteration_costs: Liste des cots  chaque itration
        best_costs: Liste des meilleurs cots  chaque itration
    """
    plt.figure(figsize=(12, 6))
    plt.plot(iteration_costs, 'b', alpha=0.5, label='Cot courant')
    plt.plot(best_costs, 'r-', linewidth=2, label='Meilleur cot')
    plt.xlabel('Itration')
    plt.ylabel('Cot')
    plt.title('volution du cot pendant la recherche tabou')
    plt.legend()
    plt.grid(True)
    plt.show()
    
def plot_cost_evolution_longueur(Longueurs, best_costs):
    """
    Trace l'volution des cots pour diffrentes longueur de list tabou.

    """
    if len(Longueurs) != len(best_costs):
        raise ValueError("Les listes Longueurs et cot doivent avoir la mme longueur.")

    plt.plot(Longueurs,best_costs, 'r',markersize=8, label='Meilleur cot')
    plt.xlabel('Longueur')
    plt.ylabel('Cot')
    plt.title('volution du cot avec la longueur de la liste tabou')
    plt.legend()
    plt.grid(True)
    plt.show()
