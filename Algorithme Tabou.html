q = 40      # Capacité maximale du camion
omega = 50  # Coût fixe par camion utilisé

def fitness(state, solution, distance_matrix):
    """
    Calcule la fonction objectif d'une solution donnée.
    
    Args:
        state: Dictionnaire contenant les informations du problème
        solution: Liste d'entiers représentant l'ordre de visite des clients
        distance_matrix: Matrice des distances entre les points
        
    Returns:
        Valeur de la fonction objectif (coût total)
    """
    orders = state["orders"]

    delivered = set()
    capacity = q
    total_distance = 0
    
    penalty_count = 0  # Compteur de violations de contraintes
    penalty_factor = 1000  # Multiplicateur de pénalité

    trucks = 0

    for i in range(1, len(solution)):
        prev, curr = solution[i-1], solution[i]
        
        # Vérifier si le chemin existe
        if distance_matrix[prev, curr] == float("inf"):
            penalty_count += 6  # Le chemin n'existe pas

        total_distance += distance_matrix[prev, curr]
        
        # Si on est chez un client, livrer sa commande
        if curr != 0:  
            if capacity < orders[curr]:
                penalty_count += (orders[curr] - capacity + 1) // 2  # Capacité insuffisante
            
            capacity -= orders[curr]
            delivered.add(curr)
        
        # Si on est de retour au dépôt, recharger le camion
        if curr == 0:
            trucks += 1
            capacity = q

    # S'assurer que toutes les commandes sont livrées
    if len(delivered) < len(orders) - 1:  # En excluant le dépôt
        penalty_count += 5 * (len(orders) - 1 - len(delivered))  # Certaines commandes n'ont pas été livrées

    # S'assurer que le chemin commence et se termine au dépôt
    if solution[0] != 0 or solution[-1] != 0:
        penalty_count += 2  # La route ne commence/finit pas au dépôt

    # Score final: distance totale + coût des camions + pénalités
    trucks_value = trucks * omega
    penalty_value = penalty_count * penalty_factor
    return total_distance + trucks_value + penalty_value

def compute_distance_matrix(state):
    """
    Calcule la matrice des distances entre tous les points.
    
    Args:
        state: Dictionnaire contenant les positions des points
        
    Returns:
        Matrice des distances
    """
    coords = state["position"]
    n = len(coords)

    dist_matrix = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            dist_matrix[i, j] = np.sqrt((coords[i][0] - coords[j][0])**2 + (coords[i][1] - coords[j][1])**2)
    return dist_matrix

def construct_initial_solution():
    """
    Retourne la solution initiale fournie sous forme de liste plate.
    
    Returns:
        Liste représentant la solution initiale
    """
    return [0, 1, 2, 3, 0, 4, 5, 6, 0, 7, 8, 9, 10, 11, 12, 0]

def count_trucks(solution):
    """
    Compte le nombre de camions utilisés (nombre de retours au dépôt).
    
    Args:
        solution: Liste représentant une solution
        
    Returns:
        Nombre de camions utilisés
    """
    return solution.count(0) - 1  # On retire le premier 0 (départ)

def is_valid_solution(solution, orders, capacity):
    """
    Vérifie que chaque camion respecte sa capacité.
    
    Args:
        solution: Liste représentant une solution
        orders: Liste des commandes des clients
        capacity: Capacité maximale du camion
        
    Returns:
        Booléen indiquant si la solution est valide
    """
    total_load = 0
    for node in solution:
        if node == 0:  # Retour au dépôt -> Réinitialisation de la charge
            total_load = 0
        else:
            total_load += orders[node]
            if total_load > capacity:
                return False  # Dépassement de la capacité
    return True

L'algorithme de recherche tabou est une métaheuristique qui permet d'éviter les optima locaux en interdisant temporairement (mettant "tabou") certains mouvements qui pourraient ramener à des solutions déjà visitées.

Notre implémentation :

  1. Commence avec une solution initiale.
  2. Explore le voisinage en effectuant des échanges entre les positions des clients.
  3. Choisit le meilleur voisin non tabou (ou utilise un critère d'aspiration).
  4. Met à jour la liste tabou et continue jusqu'à atteindre le nombre maximum d'itérations.

La taille de la liste tabu_list varie au cours des itérations, mais elle est principalement contrôlée par le paramètre tabu_tenure. Voici son évolution :

  1. Ajout d'un mouvement tabou :

    • Lorsqu'un mouvement (i, j) est sélectionné pour une nouvelle solution, il est ajouté à tabu_list :
      tabu_list.append(move)

  2. Limitation de la taille de la liste tabou :

    • Si la taille de tabu_list dépasse tabu_tenure, le plus ancien mouvement est supprimé (stratégie FIFO) :
      if len(tabu_list) > tabu_tenure:
    tabu_list.pop(0)

  3. Réinitialisation conditionnelle :

    • Si une meilleure solution globale est trouvée, la liste tabou est vidée :
      tabu_list = []

Conclusion :¶

  • En général, tabu_list ne dépasse pas tabu_tenure éléments, sauf lorsqu'elle est vidée après l'amélioration d'une solution.
  • Si aucune meilleure solution globale n'est trouvée pendant max_iter, la taille de tabu_list sera au plus tabu_tenure après quelques itérations.
  • Si une nouvelle meilleure solution est trouvée, tabu_list est réinitialisée et repart de zéro.

def tabu_search(state, initial_solution, max_iter, tabu_tenure):
    """
    Implémentation de la recherche tabou pour le problème de tournées de véhicules.
    
    Args:
        state: Dictionnaire contenant les informations du problème
        initial_solution: Solution initiale
        max_iter: Nombre maximum d'itérations
        tabu_tenure: Taille de la liste tabou
        
    Returns:
        Tuple contenant la meilleure solution trouvée et son coût
    """
    start_time = time.time()  # Début du chronomètre
    
    distance_matrix = compute_distance_matrix(state)
    best_solution = initial_solution.copy()
    current_solution = initial_solution.copy()
    best_cost = fitness(state, best_solution, distance_matrix)
    current_cost = best_cost
    tabu_list = []
    
    # Pour suivre l'évolution de la recherche
    iteration_costs = []
    best_costs = []
    
    for iteration in range(max_iter):
        neighborhood = []

        # Générer des voisins avec des swaps
        for i in range(1, len(current_solution) - 1):  # Ignorer le dépôt (0)
            for j in range(i+1, len(current_solution) - 1):
                if i != j:
                    new_solution = current_solution.copy()
                    new_solution[i], new_solution[j] = new_solution[j], new_solution[i]
                    
                    if is_valid_solution(new_solution, state['orders'], q):
                        cost = fitness(state, new_solution, distance_matrix)
                        move = (i, j)
                        neighborhood.append((new_solution, cost, move))
        
        if not neighborhood:
            print(f"Itération {iteration}: Aucun voisin valide trouvé, arrêt de la recherche.")
            break  # Si aucun voisin n'est valide, on arrête

        # Trier les voisins par coût
        neighborhood.sort(key=lambda x: x[1])

        # Sélectionner le meilleur voisin non tabou (ou le meilleur voisin si aspiration)
        found_better = False
        for candidate, cost, move in neighborhood:
            if move not in tabu_list or cost < best_cost:  # Critère d'aspiration
                current_solution = candidate.copy()
                current_cost = cost
                
                if move not in tabu_list:
                    tabu_list.append(move)
                    if len(tabu_list) > tabu_tenure:
                        tabu_list.pop(0)
                
                if cost < best_cost:
                    best_solution = candidate.copy()
                    best_cost = cost
                
                found_better = True
                break
        
        if not found_better:
            # Si tous les mouvements sont tabou, on choisit le meilleur quand même
            candidate, cost, move = neighborhood[0]
            current_solution = candidate.copy()
            current_cost = cost
            tabu_list.append(move)
            if len(tabu_list) > tabu_tenure:
                tabu_list.pop(0)
        
        # Enregistrer les coûts pour l'analyse
        iteration_costs.append(current_cost)
        best_costs.append(best_cost)
        
       
    
    end_time = time.time()  # Fin du chronomètre
    execution_time = end_time - start_time
    
    
    # Retourner la meilleure solution, son coût, et les données d'analyse
    return best_solution, best_cost, iteration_costs, best_costs, execution_time

def plot_solution(state, solution, title=""):
    """
    Visualise la solution sur un graphique.
    
    Args:
        state: Dictionnaire contenant les positions des points
        solution: Liste représentant une solution
        title: Titre du graphique
    """
    positions = state["position"]
    depot = positions[0]
    clients = positions[1:]
    
    plt.figure(figsize=(10, 6))
    plt.scatter(*depot, c='red', marker='s', s=100, label='Dépôt')
    for i, (x, y) in enumerate(clients, start=1):
        plt.scatter(x, y, c='blue', s=50)
        plt.text(x, y, f"{i}\n(q={state['orders'][i]})", fontsize=9, verticalalignment='bottom', horizontalalignment='right')
    
    colors = ['g', 'b', 'm', 'c', 'y', 'orange', 'purple']
    truck_count = 0
    current_color = colors[truck_count % len(colors)]
    
    for i in range(len(solution) - 1):
        start, end = solution[i], solution[i + 1]
        if start == 0 and end != 0:  # Nouveau camion qui quitte le dépôt
            truck_count += 1
            current_color = colors[(truck_count-1) % len(colors)]
            plt.plot([positions[start][0], positions[end][0]], 
                    [positions[start][1], positions[end][1]], 
                    color=current_color, linewidth=2, marker='o',
                    label=f'Camion {truck_count}')
        else:
            plt.plot([positions[start][0], positions[end][0]], 
                    [positions[start][1], positions[end][1]], 
                    color=current_color, linewidth=2, marker='o')
    
    plt.legend()
    plt.grid(True, linestyle='--', alpha=0.7)
    plt.title(f"{title} - Nombre de camions : {truck_count}")
    plt.xlabel("Coordonnée X")
    plt.ylabel("Coordonnée Y")
    plt.show()

def plot_cost_evolution_interation(iteration_costs, best_costs):
    """
    Trace l'évolution des coûts au cours des itérations.
    
    Args:
        iteration_costs: Liste des coûts à chaque itération
        best_costs: Liste des meilleurs coûts à chaque itération
    """
    plt.figure(figsize=(12, 6))
    plt.plot(iteration_costs, 'b', alpha=0.5, label='Coût courant')
    plt.plot(best_costs, 'r-', linewidth=2, label='Meilleur coût')
    plt.xlabel('Itération')
    plt.ylabel('Coût')
    plt.title('Évolution du coût pendant la recherche tabou')
    plt.legend()
    plt.grid(True)
    plt.show()
    
def plot_cost_evolution_longueur(Longueurs, best_costs):
    """
    Trace l'évolution des coûts pour différentes longueur de list tabou.

    """
    if len(Longueurs) != len(best_costs):
        raise ValueError("Les listes Longueurs et coût doivent avoir la même longueur.")

    plt.plot(Longueurs,best_costs, 'r',markersize=8, label='Meilleur coût')
    plt.xlabel('Longueur')
    plt.ylabel('Coût')
    plt.title('Évolution du coût avec la longueur de la liste tabou')
    plt.legend()
    plt.grid(True)
    plt.show()
